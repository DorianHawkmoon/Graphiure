81  sf::Time endFrame() {
82  m_time.current = m_clock.getElapsedTime();
83 
84  m_sample.accumulator -= m_sample.data[m_sample.index];
85  m_sample.data[m_sample.index] = m_time.current;
86 
87  m_sample.accumulator += m_time.current;
88  m_time.elapsed += m_time.current;
89 
90  if (++m_sample.index >= getSampleDepth()) {
91  m_sample.index = 0;
92  }
93 
94  if (m_time.current != sf::microseconds(0)) {
95  m_freq.current = 1.0f / m_time.current.asSeconds();
96  }
97 
98  if (m_sample.accumulator != sf::microseconds(0)) {
99  const float smooth = static_cast<float> (getSampleDepth());
100  m_freq.average = smooth / m_sample.accumulator.asSeconds();
101  }
102 
103  const sf::Int64 smooth = static_cast<sf::Int64> (getSampleDepth());
104  m_time.average = sf::microseconds(m_sample.accumulator.asMicroseconds() / smooth);
105 
106  if (m_freq.current < m_freq.minimum)
107  m_freq.minimum = m_freq.current;
108  if (m_freq.current > m_freq.maximum)
109  m_freq.maximum = m_freq.current;
110 
111  if (m_time.current < m_time.minimum)
112  m_time.minimum = m_time.current;
113  if (m_time.current > m_time.maximum)
114  m_time.maximum = m_time.current;
115 
116  ++m_freq.elapsed;
117 
118  return m_time.current;
119  }
