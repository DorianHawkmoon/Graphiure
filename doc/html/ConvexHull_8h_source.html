52 std::vector<Point> convertConvexHull(std::vector<Point> P) {
53  int n = P.size(), k = 0;
54  std::vector<Point> H(2 * n);
55 
56  // Sort points lexicographically
57  std::sort(P.begin(), P.end());
58 
59  // Build lower hull
60  for (int i = 0; i < n; ++i) {
61  while (k >= 2 && crossPoint(H[k - 2], H[k - 1], P[i]) <= 0) k--;
62  H[k++] = P[i];
63  }
64 
65  // Build upper hull
66  for (int i = n - 2, t = k + 1; i >= 0; i--) {
67  while (k >= t && crossPoint(H[k - 2], H[k - 1], P[i]) <= 0) k--;
68  H[k++] = P[i];
69  }
70 
71  H.resize(k);
72  return H;
73 }
