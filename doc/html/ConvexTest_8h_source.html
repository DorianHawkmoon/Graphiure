40 bool polygonIsConvex(StructCollision* collision) {
41  // For each set of three adjacent points A, B, C,
42  // find the cross product AB · BC. If the sign of
43  // all the cross products is the same, the angles
44  // are all positive or negative (depending on the
45  // order in which we visit them) so the polygon
46  // is convex.
47  bool got_negative = false;
48  bool got_positive = false;
49  //pillo todos los vértices
50  sf::VertexArray points=*(collision->vertices);
51  int num_points = points.getVertexCount();;
52  int B, C;
53  for (int A = 0; A < num_points; A++) {
54  B = (A + 1) % num_points;
55  C = (B + 1) % num_points;
56  float cross_product =
57  crossProductLength(
58  points[A].position.x, points[A].position.y,
59  points[B].position.x, points[B].position.y,
60  points[C].position.x, points[C].position.y);
61  if (cross_product < 0) {
62  got_negative = true;
63  } else if (cross_product > 0) {
64  got_positive = true;
65  }
66  if (got_negative && got_positive) return false;
67  }
68 
69  // If we got this far, the polygon is convex.
70  return true;
71 }
